void prime_number(int start,int end);

int main()
{
    int start, end;
    printf("enter start and end: ");
    scanf("%d%d",&start,&end);
    printf("prime no. between %d and %d are:",start,end);
    
     prime_number(start,end);
 

    return 0;
}

void prime_number(int start,int end)
{
    int i,flag;
    for(int i=start;i<end;i++)
    {
        if(i==1)
        continue;
        flag=1;
        
    for(int j=2;j<i/2;j++)
    {
        if(i%j==0)
       {
        flag=0;;
        break;
       }
    }
        if(flag==1)
        {
            printf("\n%d",i);
        }
        
    }
    
}
